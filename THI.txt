import socket
def get_host_info():
    try:
        host_name = socket.gethostname()
        host_ip = socket.gethostbyname(host_name)
        print(f"Hostname: {host_name}")
        print(f"IP Address: {host_ip}")
    except Exception as e:
        print(f"Khong the lay thong tin host: {e}")
if __name__ == "__main__":
    get_host_info()
